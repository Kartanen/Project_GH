def max(arg1, *args, key=lambda x: x): # arg1 â€” first in list, args â€” rest in list

    if args: # unite in list arg1 + args
        iterable = [arg1] + list(args)
    else: #if 1 elem given
        iterable = arg1
    
    result = iterable[0] #arg1==max
 
    for item in iterable[1:]: #compare max to rest of list
        if key(item) > key(result):
            result = item
    return result
